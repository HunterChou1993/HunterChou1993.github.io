                        # 第二步 新建Select和ResultType的注解

                        在annotation包下新建这两个注解

                        # ResultType

                        作用 :封装查询之后返回的数据

                        相关代码讲解

                        • 第一个注解 Retention 是指注解生命周期
                          • RUNTIME 是指在内存运行时
                        • 第二个注解 Target 是注解的使用位置
                          • METHOD 表示只能在方法上使用该注解
                        @Retention(RetentionPolicy.RUNTIME)
@Target(ElementType.METHOD)
public @interface ResultType {
    String value();
}

                        # Select

                        作用封装查询sql

                        @Retention(RetentionPolicy.RUNTIME)
@Target(ElementType.METHOD)
public @interface Select {
    String[] value();
}

                        # 第四步 新建configuration 类

                        在config 包下新建

                        /**
 * 配置类
 *  解析XM文件
 *  创建连接池
 */
public class Configuration {
    private String driver; // 驱动
    private String url; // 数据库连接地址
    private String username; // 登录账号
    private String password; // 登录密码

    // Mapper接口的全名称
    private String interName;
    // 数据库连接对象
    private DataSource dataSource;
    // 映射对象
    private Mapper mapper = new Mapper();

    // 无参构造方法
    public Configuration() {
        try {
            // 解析config.xml 文件
            SAXReader reader = new SAXReader();
            InputStream is =
                    Configuration.class.getClassLoader().getResourceAsStream(
                            "config.xml");
            Document document = reader.read(is);
            // 调用方法，将核心文件的数据加载搭配configuration类里面
            loadConfigXml(document);
            // 调用方法 初始化数据库连接对象
            createDataSource();
        }  catch (Exception e) {
            e.printStackTrace();
        }
    }

    private void createDataSource() throws Exception {
        // 创建属性集对象
        Properties properties = new Properties();
        properties.setProperty("url",url);
        properties.setProperty("username",username);
        properties.setProperty("password",password);
        properties.setProperty("driverClassName",driver);
        // 获取德鲁伊数据库连接对象
        DataSource dataSource1 = DruidDataSourceFactory.createDataSource(properties);
        // 赋值成员变量
        this.dataSource = dataSource1;
    }

    private void loadConfigXml(Document document) {
        // 查询property的所有节点
        List<Node> list = document.selectNodes("//property");
        // 遍历获取需要的数据
        for (Node node : list) {
            // 强转
            Element element = (Element) node;
            // 获取property标签的name属性值
            String name = element.attributeValue("name");
            String value = element.attributeValue("value");

            // 将value的值赋给成员变量
            switch (name) {
                case "driver":
                    this.driver = value; // 数据库驱动
                    break;
                case "url":
                    this.url = value; // 数据库地址
                    break;
                case "username":
                    this.username = value; // 数据库用户名
                    break;
                case "password":
                    this.password = value; // 数据库登录密码
                    break;
            }
        }
        // 获取UserMapper全路径
        Node node = document.selectSingleNode("//package");
        Element e = (Element) node;
        this.interName = e.attributeValue("name");
    }

    public String getInterName() {
        return interName;
    }

    public void setInterName(String interName) {
        this.interName = interName;
    }

    public Mapper getMapper() {
        return mapper;
    }

    public void setMapper(Mapper mapper) {
        this.mapper = mapper;
    }

    public DataSource getDataSource() {
        return dataSource;
    }

    public void setDataSource(DataSource dataSource) {
        this.dataSource = dataSource;
    }
}

                        # 第五步 创建sqlSession类

                        在config包里面创建

                        /*
   作用 根据 UserMapper 动态代理接口
 */
@SuppressWarnings("all")
public class SqlSession {
    // 动态代理
    public <T> T getMapper(Class<T> clazz) {
        // 1. 使用jdk中的Proxy类调用静态方法生成UserMapper的代理对象
        /*
            参数
                loader 类加载器 加载UserMapper的代理类
                interfaces 接口 UserMapper
                h 调用处理器invocationHandler
         */
        ClassLoader loader = clazz.getClassLoader();
        Class<?>[] interfaces = {clazz};
        InvocationHandler h = new InvocationHandler() {
            @Override
            public Object invoke(Object proxy, Method method, Object[] args) throws Exception {
                /*
                    2. invoke 使用代理对象 UserMapper接口中方法调用就调用该方法
                        proxy 生成的代理对象
                        method 当前执行的方法
                 */
                // 1. 创建配置类对象
                Configuration configuration = new Configuration();
                // 2. 解析UserMapper接口上两个注解
                // 获取接口
                Class<?> clazzMapper =
                        Class.forName(configuration.getInterName());
                // 获取UserMapper接口中的方法
                Method[] mapperMethods = clazz.getMethods();
                // 遍历数组取出每个方法
                for (Method mapperMethod : mapperMethods) {
                    // 判断该方法上是否含有注解
                    boolean isSelect =
                            mapperMethod.isAnnotationPresent(Select.class);
                    boolean isResultType =
                            mapperMethod.isAnnotationPresent(ResultType.class);
                    if (isSelect && isResultType) {
                        // 说明方法上有两个注解
                        Select select =
                                mapperMethod.getDeclaredAnnotation(Select.class);
                        ResultType resulType = mapperMethod.getDeclaredAnnotation(ResultType.class);
                        // 获取主键属性值
                        String[] arrSql = select.value();
                        String resultTypeValue = resulType.value();
                        // 获取Configuration类中的成员变量
                        Mapper mapper = configuration.getMapper();
                        mapper.setSql(arrSql[0]);
                        mapper.setResultType(resultTypeValue);
                    }
                }

                // 3. 调用定义方法，使用jdbc技术查询
                Connection conn = configuration.getDataSource().getConnection();
                Mapper mapper = configuration.getMapper();
                String sql = mapper.getSql();
                String resultType = mapper.getResultType();
                Class<?> resultTypePojoClazz = Class.forName(resultType);
                List list = queryForList(conn,sql,resultTypePojoClazz);
                // 将接收的数据返回
                return list;
            }
        };
        T proxyMapper = (T) Proxy.newProxyInstance(loader,interfaces,h);
        return proxyMapper;
    }

    /**
     *
     * @param conn 连接对象
     * @param sql 查询语句
     * @param resultTypePojoClazz 返回值实体类
     * @return 返回ArrayList对象
     */
    private List queryForList(Connection conn, String sql,
                              Class<?> resultTypePojoClazz) throws Exception {
        // 1. 创建集合对象
        ArrayList list = new ArrayList();
        // 2. 估计连接对象调用方法获取预编译对象
        PreparedStatement pst = conn.prepareStatement(sql);
        // 3. 发送sql语句
        ResultSet resultSet = pst.executeQuery();
        // 4. 处理结果集合
        while (resultSet.next()){
            // 有数据
            // 获取数据
            // 使用反射获取类中的所有成员变量
            Field[] fields = resultTypePojoClazz.getDeclaredFields();
            // 创建反射对象
            Object user = resultTypePojoClazz.getConstructor().newInstance();
            for (Field field : fields) {
                // 暴力反射
                field.setAccessible(true);
                // 获取成员变量名
                String name = field.getName();
                // 根据名字获取数据
                Object fieldValue = resultSet.getObject(name);
                // 将获取的字段值赋值给User实体类
                field.set(user,fieldValue);
            }
            // 将对象添加到集合里面
            list.add(user);
        }
        return list;
    }
}
