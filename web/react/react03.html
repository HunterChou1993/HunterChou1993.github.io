                                # 03 react的for循环

                                特点

                                1. 了解在react是如何使用for循环的
                                2. vue和react的for循环有什么不一样
                                3. react的for循环有几种实现方式

                                # 案例二 电影列表实现方式一

                                • 利用js中的for进行循环
                                  <body>
    <div id="root"></div>

    <script src="../lib/react.js"></script>
    <script src="../lib/react-dom.js"></script>
    <script src="../lib/babel.js"></script>

    <script type="text/babel">
      const root = ReactDOM.createRoot(document.querySelector('#root'))

      class App extends React.Component {
        constructor() {
          super()

          this.state = {
            movies: ['星际穿越', '流浪地球', '碟中谍', '火星救援'],
          }
        }

        // 渲染
        render() {
          // 1. 方法一
          const liEls = []
          for (let i = 0; i < this.state.movies.length; i++) {
            const movie = this.state.movies[i]
            const liEl = <li>{movie}</li>
            liEls.push(liEl)
          }
          return (
            <div>
              <h2>电影列表</h2>
              <ul>{liEls}</ul>
            </div>
          )
        }
      }

      root.render(<App />)
    </script>
  </body>

                                # 方式二

                                • 利用map进行循环
                                  <body>
    <div id="root"></div>

    <script src="../lib/react.js"></script>
    <script src="../lib/react-dom.js"></script>
    <script src="../lib/babel.js"></script>

    <script type="text/babel">
      const root = ReactDOM.createRoot(document.querySelector('#root'))

      class App extends React.Component {
        constructor() {
          super()

          this.state = {
            movies: ['星际穿越', '流浪地球', '碟中谍', '火星救援'],
          }
        }

        // 渲染
        render() {
          // 2. 方法二
          const liEls = this.state.movies.map((movie) => <li>{movie}</li>)
          return (
            <div>
              <h2>电影列表</h2>
              <ul>{liEls}</ul>
            </div>
          )
        }
      }

      root.render(<App />)
    </script>
  </body>

                                • 其实该方式跟方式二一样只是直接在return里面直接写

    <div id="root"></div>

    <script src="../lib/react.js"></script>
    <script src="../lib/react-dom.js"></script>

    <script type="text/babel">
      const root = ReactDOM.createRoot(document.querySelector('#root'))

      class App extends React.Component {
        constructor() {
          super()

          this.state = {
            movies: ['星际穿越', '流浪地球', '碟中谍', '火星救援'],
          }
        }

        // 渲染
        render() {
          return (
            <div>
              <h2>电影列表</h2>

                {this.state.movies.map((movie) => (
                  <li>{movie}</li>
                ))}
              </ul>
            </div>
          )
        }
      }

      root.render(<App />)
    </script>
  </body>
